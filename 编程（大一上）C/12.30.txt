/*Description

键盘输入5个字符串


Output

dhfghdAS
ACBdff//f

dfhjGTT/

fAdjfdfjkdf*/
#include <stdio.h>
#include<string.h>
int main()
{
   char a[5][21];
   int i,j,m,n,k;
   for(i=0;i<5;i++)
       gets(a[i]);
   for(i=0;i<5;i++)
   {
       m=0;
       for(j=0;j<strlen(a[i]);j++)
       {
           if(a[i][j]>='A'&&a[i][j]<='Z')
               m+=1;
       }
       if(i==0)
       {
          n=m;
          k=i;
       }
       else if(m<n)
       {
           k=i;
       }
 
   }
   puts(a[k]);
   return 0;
}
