、汉诺塔问题。这是一个古老的数学问题，有一个塔，上有三根柱子X，Y，Z，X柱子上有N个盘子，从下往上盘子越来越小。今欲把X柱子上的所有盘子挪到Z柱子上（可以利用B柱子），挪动过程中，只能小盘子压大盘子，不允许大盘子压小盘子。从键盘输入盘子数，请给出挪动方法和步骤。
#include<stdio.h>
void move(int n,char X,char Y,char Z)
{
	if(n>1)
	{
		move(n-1,X,Z,Y);
		printf("%c->%c\n",X,Z);
		move(n-1,Y,X,Z);
	}
	else
		printf("%c->%c\n",X,Z);
}
int main()
{
	int n;
	char X='X',Y='Y',Z='Z';
	scanf("%d",&n);
	move(n,X,Y,Z);
	return 0;
}

void draw(int n)
{
	int i,j;
	for(i=1;i<=n;i++)
	{ 
			printf("%*c",7-i,' ');
		for(j=1;j<=2*i-1;j++)
			printf("*");
		printf("\n");
