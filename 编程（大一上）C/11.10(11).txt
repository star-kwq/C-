/*Problem C: 循环：求母牛数量
Description
有一头小母牛（0岁），它从第四年（3岁）开始，每年都生一头小母牛（一年只生一头），而且，所以的小母牛也都会在第四年开始生育。假设所有的母牛都不会死，请问：第n年，此牛群共有多少头母牛？

Input
Output
Sample Input
4
8
Sample Output
2
9*/
#include<stdio.h> 
int main() 
{    
	int n,sum,a,b,c,d,i;
	scanf("%d",&n);
	if(n>=1&&n<=3);
	{
		sum=1;
	}
    if(n==4)
	{
		sum=2;
	}
	else
	{   
		n=n-4;
		a=d=1;
		b=c=0;
		for(i=0;i<n;i++)
		{    
			d=c+d;
			c=b;
			b=a;
			a=d;
		}
		sum=a+b+c+d;
	}
	printf("%d\n",sum);
    return 0; 
}

则该4位数称为分段和平方数。从键盘输入一个4位整数n，求n以内的最大平方数，若没有，输出NO。
#include<stdio.h> 
int main() 
{    
	int n,i,a,b,c,d;
	scanf("%d",&n);
	for(i=n;i>=1000;i--)
	{
		a=i/1000%10;
		b=i/100%10;
		c=i/10%10;
		d=i%10;
		if((a*10+b+c*10+d)*(a*10+b+c*10+d)==i)
			break;
	}
	if(i==999)
		printf("NO");
	else
		printf("%d\n",i);
    return 0; 
}

/*Problem B: 循环：楼梯问题（break）
Description
楼梯问题：一个楼梯，若每次跨7个台阶，则最后剩6个；若每次跨6个，最后剩5个；......，若每次跨两个，则最后剩1个，问最少可能是多少个台阶？

Input
Output
Sample Input
Sample Output
419*/
#include<stdio.h>
int main()
{
	int i;
	for(i=1;;i++)
	{
		if(i%7==6&&i%6==5&&i%5==4&&i%4==3&&i%3==2&&i%2==1)
			break;
	}
	printf("%d\n",i);
	return 0;
}

/*Problem G: 从键盘输入一行字符，统计其中英文、数字、空格及其他字符的个数。
Description
键盘输入一行字符，统计其中英文、数字、空格及其他字符的个数。

Input
一行字符

Output
英文、数字、空格及其他字符的个数。

Sample Input
hello   7890
Sample Output
5,4,3,0*/

#include<stdio.h>
int main()
{
	int b=0,c=0,d=0,e=0;
	char a;
	scanf("");
	while((a=getchar())!='\n')
	{
		if(a>='A'&&a<='Z'||a>='a'&&a<='z')
			b=b+1;
		else if(a>='0'&&a<='9')
			c=c+1;
		else if(a==32)
			d=d+1;
		else
			e=e+1;
		a++;
	}
	printf("%d,%d,%d,%d",b,c,d,e);
	return 0;
}
