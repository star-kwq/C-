/*Problem A: 循环：有一分数序列： 2/1 3/2 5/3 8/5 13/8 21/13...... 求出这个数列的前N项之和，保留两位小数。*/
#include<stdio.h>
int main()
{
	int i,n,h;
	float m=0,b=1,a=2,c;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		c=a/b;
		m=m+c;
		h=a+b;
		b=a;
		a=h;
		
	}
	printf("%.2f\n",m);
	return 0;
}

/*Problem B: 循环：自由下落问题,一球从M米高度自由下落，每次落地后返回原高度的一半，再落下。
 它在第N次落地时反弹多高？共经过多少米？ 保留两位小数*/
#include<stdio.h>
int main()
{
	int n,i;
	float g,j,m;
	scanf("%f %d",&m,&n);
	j=m;
	for(i=1;i<=n;i++)
	{
		m=m/2;
		if(i==n)
			break;
		j=j+m*2;
	}
	j=j+m;
	g=m;
	printf("%.2f %.2f\n",g,j);
	return 0;
}

/*Problem C: 循环：猴子吃桃问题。猴子第一天摘下若干个桃子，当即吃了一半，还不过瘾，又多吃了一个。 
第二天早上又将剩下的桃子吃掉一半，又多吃一个。以后每天早上都吃了前一天剩下的一半零一个。
 到第N天早上想再吃时，见只剩下一个桃子了。求第一天共摘多少桃子*/
#include<stdio.h>
int main()
{
	int i,x,m=1;
	scanf("%d",&x);
	for(i=1;i<=x;i++)
	{
		if(i==1)
			m=1;
		if(i>1)
			m=(m+1)*2;
	}
	printf("%d\n",m);
	return 0;
}

/*Problem D: 循环：输入一个正整数n的值，输出 1!-2!+3!-4!+5!.....n! 的值*/
#include<stdio.h>
int main()
{
	int i,n,m,h=0,j=0;
	float k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		if(i==1)
			m=1;
		if(i>1)
			m=m*(m+1);
		if(i%2==1)
			h+=m;
		if(i%2==0)
			j+=m;
	}
	k=h-j;
	printf("%.2f\n",k);
	return 0;
}

编制歌手大赛评分程序，评委人数及评委的打分均从键盘输入，去掉一个最高分，去掉一个最低分，求选手的最后得分（平均分），不允许使用数组。
#include<stdio.h>
int main()
{
	int n,i;
	float a,b,c,max=0,min=10,sum=0;
	scanf("%d%",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%f",&c);
		if(c>max)
		max=c;
		if(c<min)
		min=c;
		sum=sum+c;

	}
	sum=(sum-max-min)/(n-2);
	printf("%.2f",sum);
    return 0;
}
